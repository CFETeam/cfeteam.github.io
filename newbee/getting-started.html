Bee.mount('grid-view-example', {
  $data: {
    "colums": [
      {"name": "主机名", "key": "alias"},
      {"name": "所属网络", "key": "netName"},
      {"name": "内网 IP", "key": "ip"}
    ],
    count: 5,
    minHeight: 350
  },
  getData: function(options, callback) {
    setTimeout(function() {
      callback(null, {
        "list": [
          {"alias": "未命名", "netName": "基础网络", "ip": "10.143.64.70"}
        ],
        totalNum: 10
      })
    }, 0)
  }
})
